    * ### Code

        ```jsx
        import { useState } from 'react';
        import { Button } from '@rotational/beacon-core';
        export const Counter = () => {
        const [count, setCount] = useState(0);
        return (
            <div>
            <Button
                variant="primary"
                size={'small'}
                onClick={() => setCount(count + 1)}
                className={styles.counter}
            >
                Clicked {count} times
            </Button>
            </div>
        );
        };
        ```
